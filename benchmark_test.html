4. 사용 예시 코드

Dashboard에 통합할 때 사용할 수 있는 코드입니다. 

// 백테스트 데이터와 벤치마크 비교
const { equityCurve } = await Analytics.loadIntradayData(
    '2026-01-28', '2026-03-27', 1000000
);

const comparison = Benchmark.compareToBenchmark(equityCurve, 1000000);

if (comparison) {
    console.log('전략 수익률:', comparison.strategy.finalReturn.toFixed(2) + '%');
    console.log('KOSPI 수익률:', comparison.benchmark.finalReturn.toFixed(2) + '%');
    console.log('초과성과:', comparison.comparison.outperformance.toFixed(2) + '%');

    // 차트 렌더링
    const chartData = Benchmark.prepareChartData(equityCurve, 1000000);
    Charts.renderBenchmarkComparison('benchmarkChart',
        chartData.datasets[0].data.map((v, i) => ({
            date: chartData.labels[i],
            cumulativeReturn: v
        })),
        chartData.datasets[1].data.map((v, i) => ({
            date: chartData.labels[i],
            cumulativeReturn: v
        }))
    );
}
